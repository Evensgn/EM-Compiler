
func main {
%main_entry:
    jump %__init_func_entry

%__init_func_entry:
    $x = move 0
    $y = move 0
    $x = move 10
    $y = move 19
    $a = move $x
    $b = move $y
    $vreg = sgt $a $b
    br $vreg %if_then %if_after

%if_after:
    $return_value = move $b
    jump %max_end

%if_then:
    $return_value = move $a
    jump %max_end

%max_end:
    $vreg_2 = move $return_value
    ret $vreg_2

}
