
func main {
%main_entry:
    call __init_func 
    $i = move 0
    $c = move 3
    $i = move 1
    jump %for_cond

%for_cond:
    $vreg = slt $i 10
    br $vreg %for_body %for_after

%for_after:
    ret $c

%for_body:
    $c = add $c 1
    jump %for_step

%for_step:
    $vreg_2 = move $i
    $i = add $i 1
    jump %for_cond

}

func __init_func {
%__init_func_entry:
    ret 0

}
