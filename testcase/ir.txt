
func main {
%main_entry:
    jump %__init_func_entry

%__init_func_entry:
    $i = move 0
    $c = move 0
    $i = move 0
    jump %for_cond

%for_cond:
    $vreg = sle $i 1000
    br $vreg %for_body %for_after

%for_body:
    $vreg_2 = move $c
    $vreg_2 = mul $vreg_2 2
    $c = move $vreg_2
    jump %for_step

%for_step:
    $i = add $i 1
    jump %for_cond

%for_after:
    ret 0

}
