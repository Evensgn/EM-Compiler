
func main {
%main_entry:
    jump %__init_func_entry

%__init_func_entry:
    $i = move 0
    $c = move 1
    $i = move 1
    jump %for_cond

%for_cond:
    $vreg = slt $i 10
    br $vreg %for_body %for_after

%for_after:
    ret 0

%for_body:
    $c = add $c 1
    jump %for_step

%for_step:
    $i = add $i 1
    jump %for_cond

}
