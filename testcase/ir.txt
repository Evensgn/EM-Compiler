
func main {

    jump %__init_func_entry

%__init_func_entry:
    $a = move 0
    $b = move 0
    $c = move 0
    $a = move 2
    $b = move 2
    $c = move 2
    $vreg = move $b
    $vreg = add $vreg $c
    $a = move $vreg
    $vreg_2 = move $a
    $vreg_2 = sub $vreg_2 $b
    $c = move $vreg_2
    $vreg_3 = move 2
    $vreg_3 = mul $vreg_3 $a
    $vreg_4 = rem $vreg_3 $c
    $b = move $vreg_4
    ret 0

}
